"use client";

import React, { useEffect, useState } from "react";
import {
  User,
  Mail,
  Phone,
  Calendar,
  AlertCircle,
  CheckCircle2,
  Users,
} from "lucide-react";
import { stateOptions, getCitiesByState } from "../../../data/statesCities";

// Import Customer API functions
import { createCustomerApi } from "../../features/Customer/customerApi";

interface CustomerRecord {
  firstName: string;
  lastName: string;
  email: string;
  phone: string;
  dateOfBirth: string;
  anniversary: string;
  gender: string;
  address: string;
  state: string;
  city: string;
  pincode: string;
}

// Initial empty state
const initialFormState: CustomerRecord = {
  firstName: "",
  lastName: "",
  email: "",
  phone: "",
  dateOfBirth: "",
  anniversary: "",
  gender: "",
  address: "",
  state: "",
  city: "",
  pincode: "",
};

const CustomerForm = () => {
  const [customerFormData, setFormData] =
    useState<CustomerRecord>(initialFormState);
  const [touchedFields, setTouchedFields] = useState<Record<string, boolean>>(
    {},
  );
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [cities, setCities] = useState<string[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [isSuccess, setIsSuccess] = useState(false);

  const genderOptions = ["Male", "Female", "Other"];

  // Update cities when state changes
  useEffect(() => {
    if (customerFormData.state) {
      const cityList = getCitiesByState(customerFormData.state);
      setCities(cityList);
      // Reset city if current city not in new city list
      if (customerFormData.city && !cityList.includes(customerFormData.city)) {
        setFormData((prev) => ({ ...prev, city: "" }));
      }
    } else {
      setCities([]);
    }
  }, [customerFormData.state, customerFormData.city]);

  // Reset form after successful submission
  useEffect(() => {
    if (isSuccess) {
      setFormData(initialFormState);
      setTouchedFields({});
      setErrors({});
      // Reset the success state after 3 seconds
      setTimeout(() => {
        setIsSuccess(false);
      }, 5000);
    }
  }, [isSuccess]);

  const handleFieldChange = (
    e: React.ChangeEvent<
      HTMLInputElement | HTMLSelectElement | HTMLTextAreaElement
    >,
  ) => {
    const { name, value } = e.target;
    setFormData((prev) => ({ ...prev, [name]: value }));
    // Clear error for this field when user starts typing
    if (errors[name]) {
      setErrors((prev) => ({ ...prev, [name]: "" }));
    }
  };

  const markFieldTouched = (fieldName: string) => {
    setTouchedFields((prev) => ({ ...prev, [fieldName]: true }));
  };

  const validateForm = (): boolean => {
    const newErrors: Record<string, string> = {};

    const requiredFields: Array<keyof CustomerRecord> = [
      "firstName",
      "lastName",
      "email",
      "phone",
      "address",
      "state",
      "city",
      "pincode",
    ];

    requiredFields.forEach((field) => {
      if (!customerFormData[field]?.trim()) {
        newErrors[field] = `${
          field === "firstName"
            ? "First name"
            : field === "lastName"
              ? "Last name"
              : field
        } is required`;
      }
    });

    if (
      customerFormData.email &&
      !/\S+@\S+\.\S+/.test(customerFormData.email)
    ) {
      newErrors.email = "Enter valid email";
    }

    if (customerFormData.phone && !/^\d{10}$/.test(customerFormData.phone)) {
      newErrors.phone = "Phone must be 10 digits";
    }

    if (customerFormData.pincode && !/^\d{6}$/.test(customerFormData.pincode)) {
      newErrors.pincode = "Pincode must be 6 digits";
    }

    // Optional date validations
    if (customerFormData.dateOfBirth) {
      const dob = new Date(customerFormData.dateOfBirth);
      const today = new Date();
      if (dob > today) {
        newErrors.dateOfBirth = "Date of birth cannot be in future";
      }
    }

    if (customerFormData.anniversary) {
      const anniversary = new Date(customerFormData.anniversary);
      const today = new Date();
      if (anniversary > today) {
        newErrors.anniversary = "Anniversary cannot be in future";
      }
    }

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    // Mark all fields touched
    const allTouched: Record<string, boolean> = {};
    Object.keys(customerFormData).forEach((k) => (allTouched[k] = true));
    setTouchedFields(allTouched);

    // Validate form first
    if (!validateForm()) {
      return;
    }

    try {
      setIsLoading(true);
      setError(null);

      await createCustomerApi(customerFormData); // ✅ only one call

      setIsSuccess(true);

      window.scrollTo({
        top: 0,
        behavior: "smooth",
      });

      setFormData(initialFormState); // reset form
    } catch (err: any) {
      console.error("Registration failed:", err);
      setError(err?.message || "Registration failed");
    } finally {
      setIsLoading(false);
    }
  };

  const getInputClass = (field: string, icon = true) => {
    const base = `w-full ${
      icon ? "pl-10" : "px-4"
    } pr-4 py-3 border rounded-lg focus:outline-none focus:ring-2 transition-colors`;

    if (touchedFields[field] && errors[field]) {
      return `${base} border-red-500 bg-red-50 focus:ring-red-200`;
    }
    return `${base} border-gray-300 focus:ring-blue-200 focus:border-blue-400`;
  };

  // Get error message
  const getErrorMessage = () => {
    return error;
  };

  return (
    <div className="w-full bg-gray-50">
      <div className="sticky top-0 z-30 bg-orange-100 p-3 rounded-md shadow-sm">
        <div className="flex justify-between items-center">
          <div className="pl-4 border-l-8 border-orange-500 bg-white px-3 rounded-md shadow-md">
            <h2 className="text-4xl font-bold text-left py-4 text-orange-600">
              Customer Registration Form
            </h2>
          </div>
        </div>
      </div>

      <div className="p-6 w-full  mx-auto bg-white shadow-xl rounded-lg my-6">
        {/* Status Messages */}
        {(error || isSuccess) && (
          <div className="mb-6">
            {error && (
              <div className="flex items-center gap-2 p-4 text-red-700 bg-red-50 rounded-lg border border-red-200">
                <AlertCircle size={20} />
                <span className="font-medium">{getErrorMessage()}</span>
              </div>
            )}
            {isSuccess && (
              <div className="fixed top-6 right-6 z-50 flex items-center gap-2 px-5 py-3 text-green-700 bg-green-50 border border-green-300 rounded-lg shadow-lg animate-slideIn">
                <CheckCircle2 size={20} />
                <span className="font-medium">
                  Customer registered successfully!
                </span>
              </div>
            )}
          </div>
        )}

        <form onSubmit={handleSubmit} className="space-y-6">
          {/* PERSONAL INFORMATION SECTION */}
          <div className="p-6 border rounded-xl bg-purple-50">
            <h3 className="text-xl font-semibold text-blue-800 mb-6 pb-3 border-b relative">
              <span className="bg-blue-600 text-white px-3 py-1 rounded-md mr-2">
                1
              </span>
              Personal Information
            </h3>

            <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
              <Input
                label="First Name"
                name="firstName"
                icon={<User size={18} />}
                value={customerFormData.firstName}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.firstName}
                touched={touchedFields.firstName}
                getInputClass={getInputClass}
                required
              />

              <Input
                label="Last Name"
                name="lastName"
                icon={<User size={18} />}
                value={customerFormData.lastName}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.lastName}
                touched={touchedFields.lastName}
                getInputClass={getInputClass}
                required
              />

              <Input
                label="Email"
                name="email"
                type="email"
                icon={<Mail size={18} />}
                value={customerFormData.email}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.email}
                touched={touchedFields.email}
                getInputClass={getInputClass}
                required
              />

              <Input
                label="Phone"
                name="phone"
                type="tel"
                icon={<Phone size={18} />}
                maxLength={10}
                value={customerFormData.phone}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.phone}
                touched={touchedFields.phone}
                getInputClass={getInputClass}
                required
                placeholder="10-digit mobile number"
              />

              <Input
                label="Date of Birth"
                name="dateOfBirth"
                type="date"
                icon={<Calendar size={18} />}
                value={customerFormData.dateOfBirth}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.dateOfBirth}
                touched={touchedFields.dateOfBirth}
                getInputClass={getInputClass}
              />

              <Input
                label="Anniversary"
                name="anniversary"
                type="date"
                icon={<Calendar size={18} />}
                value={customerFormData.anniversary}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.anniversary}
                touched={touchedFields.anniversary}
                getInputClass={getInputClass}
              />

              <SelectField
                label="Gender"
                name="gender"
                icon={<Users size={18} />}
                options={genderOptions}
                value={customerFormData.gender}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                getInputClass={getInputClass}
              />
            </div>
          </div>

          {/* ADDRESS INFORMATION SECTION */}
          <div className="p-6 border rounded-xl bg-green-50">
            <h3 className="mb-4 text-xl font-semibold text-green-800 flex items-center gap-2">
              <svg
                xmlns="http://www.w3.org/2000/svg"
                width="20"
                height="20"
                viewBox="0 0 24 24"
                fill="none"
                stroke="currentColor"
                strokeWidth="2"
                strokeLinecap="round"
                strokeLinejoin="round"
              >
                <path d="M20 10c0 6-8 12-8 12s-8-6-8-12a8 8 0 0 1 16 0Z" />
                <circle cx="12" cy="10" r="3" />
              </svg>
              Address Information
            </h3>

            <div className="mb-4">
              <label className="block font-medium text-gray-700 mb-1">
                Address <span className="text-red-500">*</span>
              </label>
              <textarea
                name="address"
                placeholder="Enter complete address"
                value={customerFormData.address}
                onChange={handleFieldChange}
                onBlur={() => markFieldTouched("address")}
                className={getInputClass("address", false)}
                rows={3}
                required
              />
              {touchedFields.address && errors.address && (
                <p className="text-sm text-red-600 mt-1">{errors.address}</p>
              )}
            </div>

            <div className="grid gap-4 md:grid-cols-3">
              <div>
                <label className="block font-medium text-gray-700 mb-1">
                  City <span className="text-red-500">*</span>
                </label>
                <input
                  list="cities"
                  name="city"
                  placeholder="Select or type city"
                  value={customerFormData.city}
                  onChange={handleFieldChange}
                  onBlur={() => markFieldTouched("city")}
                  className={getInputClass("city", false)}
                  required
                  disabled={!customerFormData.state}
                />
                <datalist id="cities">
                  {cities.map((c) => (
                    <option key={c} value={c} />
                  ))}
                </datalist>
                {touchedFields.city && errors.city && (
                  <p className="text-sm text-red-600 mt-1">{errors.city}</p>
                )}
              </div>
              <div>
                <label className="block font-medium text-gray-700 mb-1">
                  State <span className="text-red-500">*</span>
                </label>
                <input
                  list="states"
                  name="state"
                  placeholder="Select or type state"
                  value={customerFormData.state}
                  onChange={handleFieldChange}
                  onBlur={() => markFieldTouched("state")}
                  className={getInputClass("state", false)}
                  required
                />
                <datalist id="states">
                  {stateOptions.map((s) => (
                    <option key={s} value={s} />
                  ))}
                </datalist>
                {touchedFields.state && errors.state && (
                  <p className="text-sm text-red-600 mt-1">{errors.state}</p>
                )}
              </div>

              <div>
                <label className="block font-medium text-gray-700 mb-1">
                  Pincode <span className="text-red-500">*</span>
                </label>
                <input
                  name="pincode"
                  type="text"
                  placeholder="6-digit pincode"
                  maxLength={6}
                  value={customerFormData.pincode}
                  onChange={handleFieldChange}
                  onBlur={() => markFieldTouched("pincode")}
                  className={getInputClass("pincode", false)}
                  required
                />
                {touchedFields.pincode && errors.pincode && (
                  <p className="text-sm text-red-600 mt-1">{errors.pincode}</p>
                )}
              </div>
            </div>
          </div>

          {/* SUBMIT BUTTON */}
          <div className="flex justify-center pt-4">
            <button
              type="submit"
              disabled={isLoading}
              className="px-10 py-3 text-white bg-blue-600 rounded-full hover:bg-blue-700 
                       disabled:opacity-50 disabled:cursor-not-allowed transition-all
                       font-semibold text-lg shadow-md hover:shadow-lg"
            >
              {isLoading ? (
                <span className="flex items-center gap-2">
                  <svg className="animate-spin h-5 w-5" viewBox="0 0 24 24">
                    <circle
                      className="opacity-25"
                      cx="12"
                      cy="12"
                      r="10"
                      stroke="currentColor"
                      strokeWidth="4"
                      fill="none"
                    />
                    <path
                      className="opacity-75"
                      fill="currentColor"
                      d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"
                    />
                  </svg>
                  Saving...
                </span>
              ) : (
                "Register Customer"
              )}
            </button>
          </div>
        </form>
      </div>
    </div>
  );
};

// Reusable Input component
const Input = ({
  label,
  name,
  type = "text",
  icon,
  value,
  onChange,
  onBlur,
  error,
  touched,
  getInputClass,
  maxLength,
  required = false,
  placeholder,
}: any) => (
  <div>
    <label className="block font-medium text-gray-700 mb-1">
      {label} {required && <span className="text-red-500">*</span>}
    </label>
    <div className="relative">
      <input
        type={type}
        name={name}
        value={value}
        onChange={onChange}
        onBlur={() => onBlur(name)}
        className={getInputClass(name, true)}
        maxLength={maxLength}
        placeholder={placeholder}
      />
      <span className="absolute left-3 top-3 text-gray-400">{icon}</span>
    </div>
    {touched && error && <p className="text-sm text-red-600 mt-1">{error}</p>}
  </div>
);

// Reusable Select component
const SelectField = ({
  label,
  name,
  icon,
  options,
  value,
  onChange,
  onBlur,
  getInputClass,
}: any) => (
  <div>
    <label className="block font-medium text-gray-700 mb-1">{label}</label>
    <div className="relative">
      <select
        name={name}
        value={value}
        onChange={onChange}
        onBlur={() => onBlur(name)}
        className={getInputClass(name, true)}
      >
        <option value="">Select {label}</option>
        {options.map((o: string) => (
          <option key={o} value={o}>
            {o}
          </option>
        ))}
      </select>
      <span className="absolute left-3 top-3 text-gray-400">{icon}</span>
    </div>
  </div>
);

export default CustomerForm;
